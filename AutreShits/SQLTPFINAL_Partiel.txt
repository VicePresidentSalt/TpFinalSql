create table Divisions
(
  NomDivision varchar(40) constraint PKNOMDIV Primary Key,
  DateCreation date
);

create table Equipes
(
  NomEquipe varchar(40) constraint PKNOMEQUIPE Primary Key,
  DateIntroLigue Date,
  LogoEquipe blob,
  DivisionEquipe varchar(40),
  VilleEquipe varchar(40),
  constraint FKDIV Foreign Key(DivisionEquipe) References Division(NomDivision)
);

create table Joueurs
(
  NomJoueur varchar(40),
  PrenomJoueur varchar(40),
  DateNaissance date,
  NumeroJoueur Number,
  PhotoJoueur blob,
  EquipeJoueur varchar(40),
  PositionJoueur varchar(40),
  constraint FKEquipe Foreign Key(EquipeJoueur) References Equipes(NomEquipe)
  
);
